### Task 2: 挂件类 UIPiggyBankHudEntry
**Files:** Create `client/Assets/Scripts/UI/Gacha/UIPiggyBankHudEntry.cs`

要点（完整实现照此写，动手前先对照 `UI/Public/UIBtnGift.cs` 确认 UIWidget 基类的 `mGo`/`Show/Hide/IsActive` 语义）：

```csharp
using System.Collections.Generic;
using GameCommon;
using GameCommon.Cfg;
using GameCommon.Const;
using GameEvent;
using Logic;
using TFW;
using UnityEngine;
using UnityEngine.EventSystems;

namespace UI
{
    /// <summary>
    /// 异国美酒储蓄罐 HUD 挂件（挂 UIHeroLottery 下方）。
    /// 可买(有档今日UTC未砸)=图标+红点(当日点过即熄,次日重亮)；均CD=图标+倒计时到明日UTC0点。
    /// 附带：进界面自动弹获取途径(有档可买&&美酒=0,每登录一次) + BI打点。
    /// </summary>
    public partial class UIPiggyBankHudEntry : UIWidget
    {
        protected override void OnLoad()
        {
            mGoRedPoint = mGo.transform.Find("RedPoint").gameObject;
            mTxtCd = mGo.transform.Find("TxtCd").GetComponent<TFWText>();
        }

        protected override void RegisterEvent()
        {
            UIHelper.AddListener(TFW.EventTriggerType.Click, mGo, OnClickTrigger);
            EventMgr.I.RegisterEvent(TEventType.RefreshPackGiftUI, OnRefreshPackGiftUI, this);
        }

        protected override void UnregisterEvent()
        {
            UIHelper.RemoveListener(TFW.EventTriggerType.Click, mGo);
            EventMgr.I.UnregisterEvent(this);
        }

        private void OnRefreshPackGiftUI(long packID) => RefreshState();   // 砸罐后自动切态

        /// <summary>宿主 RefreshView 每次切池调用。</summary>
        public void Refresh(CDrawCardsCfg gachaCfg)
        {
            mVisible = gachaCfg != null && gachaCfg.CostType == GiftConst.DAILY_RESET_PIGGY_RES_ID && CollectTiers();
            mGo.SetActive(mVisible);
            if (!mVisible) { UIHelper.StopTiming(mTxtCd); return; }
            mCostNum = gachaCfg.CostNum;
            RefreshState();
            TryReportShow();      // 每次界面OnShown后首个Refresh报一次曝光
            TryAutoPopup();       // 弹窗检查(每登录一次)
        }

        public void OnHostShown() => mShowReported = false;  // 宿主OnShown时调,重置曝光去重
        public void OnHostHidden() { UIHelper.StopTiming(mTxtCd); Hide(); }

        // —— 状态判定(全UTC自然日口径,同 UIPiggyBankContent.IsDailyResetPiggy) ——
        private bool CollectTiers()
        {
            mTierPackIDs.Clear();
            var giftMeta = G.Player.GetMeta<GiftMeta>();
            if (!CPiggyBank.Instance.GroupInfo.TryGetValue(GiftConst.DAILY_RESET_PIGGY_RES_ID, out var groupInfo)) return false;
            foreach (var cfgID in groupInfo.Vals)
            {
                var packID = CPiggyBank.I(cfgID).PackID;
                if (giftMeta.CheckCanShowGift(packID)) mTierPackIDs.Add(packID);
            }
            return mTierPackIDs.Count > 0;  // 全不可见(如等级<3)→隐藏
        }

        private bool AnyTierBuyable()
        {
            var giftMeta = G.Player.GetMeta<GiftMeta>();
            var todayUtc = GameTime.GetUtcDateTime().Date;
            foreach (var packID in mTierPackIDs)
            {
                var bt = giftMeta.GetGiftBuyingTime(packID);
                var inCd = bt > 0 && GameTime.GetUtcDateTime(bt).Date == todayUtc;
                if (!inCd) return true;
            }
            return false;
        }

        private void RefreshState()
        {
            if (!mVisible) return;
            UIHelper.StopTiming(mTxtCd);
            if (AnyTierBuyable())
            {
                mTxtCd.gameObject.SetActive(false);
                mGoRedPoint.SetActive(!ClickedToday());
            }
            else  // 均CD→倒计时到明日UTC0点,走完自动切回
            {
                mGoRedPoint.SetActive(false);
                mTxtCd.gameObject.SetActive(true);
                var nowUtc = GameTime.GetUtcDateTime();
                var endTime = GameTime.Time + (long)(nowUtc.Date.AddDays(1) - nowUtc).TotalMilliseconds;
                UIHelper.StartTiming(mTxtCd, endTime, onEnd: RefreshState);
            }
        }

        // —— 红点按日点击记录(按角色隔离,范式=UIHelper.Update.cs 红包弹窗) ——
        private static int TodayIndex() => (int)(GameTime.UtcTodayStartTimestamp / 86400000L);
        private static bool ClickedToday() => GamePlayerPrefs.GetIntByRole(CLICK_DATE_KEY, 0) == TodayIndex();

        private void OnClickTrigger(GameObject go, PointerEventData data)
        {
            GamePlayerPrefs.SetByRole(CLICK_DATE_KEY, TodayIndex());
            mGoRedPoint.SetActive(false);
            TgsHelper.Instance.TraceUserClick(BIConst.BI_PIGGY_HUD_CLICK);
            UIHelper.ShowItemObtain(GiftConst.DAILY_RESET_PIGGY_RES_ID, mCostNum);
        }

        // —— BI曝光(每次OnShown一次) ——
        private void TryReportShow()
        {
            if (mShowReported) return;
            mShowReported = true;
            TgsHelper.Instance.TraceApplicationLog(BIConst.BI_PIGGY_HUD_SHOW,
                new Dictionary<string, string> { { "state", AnyTierBuyable() ? "buyable" : "cd" } });
        }

        // —— 自动弹窗(每登录一次;G.Player 每次登录重建,引用变=新登录。动手时验证该假设,不成立改挂登录事件重置) ——
        private void TryAutoPopup()
        {
            if (ReferenceEquals(sPopupShownPlayer, G.Player)) return;
            if (!AnyTierBuyable()) return;
            if (G.Player.GetMeta<StorageMeta>().GetItemNum(GiftConst.DAILY_RESET_PIGGY_RES_ID) > 0) return;
            sPopupShownPlayer = G.Player;
            TgsHelper.Instance.TraceApplicationLog(BIConst.BI_PIGGY_OBTAIN_POPUP);
            UIHelper.ShowItemObtain(GiftConst.DAILY_RESET_PIGGY_RES_ID, mCostNum);
        }

        private const string CLICK_DATE_KEY = "PiggyHudClick_Date";
        private static object sPopupShownPlayer;
        private readonly List<int> mTierPackIDs = new();
        private GameObject mGoRedPoint;
        private TFWText mTxtCd;
        private bool mVisible;
        private bool mShowReported;
        private int mCostNum = 1;
    }
}
```

注意核对项（写完逐个确认，不对就按真实 API 调整）：
- `UIWidget` 基类名/`mGo` 字段名以 UIBtnGift 实际继承为准；`TFWText`/`TgsHelper`/`GamePlayerPrefs`/`GameTime.UtcTodayStartTimestamp` 的命名空间照红包弹窗那段抄。
- `TraceUserClick` 参数形态照 `UiActivityCommonRules.cs:138`。
- 若 UIWidget 无 partial Auto 要求，此类不需要 Auto 文件（UIBtnGift 的 Auto 是空壳）。
